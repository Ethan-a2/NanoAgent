GRPO 核心算法

损失函数公式

# GRPO Loss (简化版)
L_GRPO = E[min(
    r_θ(x, y) * A,
    clip(r_θ(x, y), 1-ε, 1+ε) * A
)]

其中:
- r_θ(x, y) = π_θ(y|x) / π_ref(y|x)
- A = 优势函数 (Advantage)
- ε = 裁剪参数 (通常为0.2)

关键实现

# 优势函数计算（grpo-torch.py）
advantages = rewards - baseline
advantages = (advantages - advantages.mean()) / \
             (advantages.std() + 1e-8)

# 重加权机制
weights = compute_group_weights(advantages)
loss = (log_probs * advantages * weights).mean()

核心工具函数示例

# utils/tools.py - AverageMeter
class AverageMeter:
    def __init__(self):
        self.reset()
    
    def reset(self):
        self.val = 0
        self.avg = 0
        self.sum = 0
        self.count = 0
    
    def update(self, val, n=1):
        self.val = val
        self.sum += val * n
        self.count += n
        self.avg = self.sum / self.count

推理系统

notebooks/inference.py

class NanoAgentInference:
    def __init__(self, model_path, tokenizer_path):
        self.model = AutoModel.from_pretrained(model_path)
        self.tokenizer = AutoTokenizer.from_pretrained(tokenizer_path)
        self.model.eval()
    
    def generate(self, prompt, max_length=512):
        inputs = self.tokenizer(prompt, return_tensors="pt")
        with torch.no_grad():
            outputs = self.model.generate(
                **inputs,
                max_length=max_length,
                temperature=0.7,
                top_p=0.9
            )
        return self.tokenizer.decode(outputs[0])
    
    def batched_generate(self, prompts):
        # 批量推理优化
        return [self.generate(p) for p in prompts]

  • 模型大小: 参数量和显存
# benchmarks/bfcl/bfcl_eval.py
def evaluate_model(model, test_cases):
    results = []
    for case in test_cases:
        prediction = model.generate(case['input'])
        score = compute_metrics(prediction, case['target'])
        results.append(score)
    
    return {
        'accuracy': np.mean([r['exact_match'] for r in results]),
        'f1_score': np.mean([r['f1'] for r in results]),
        'avg_latency': np.mean([r['latency'] for r in results])
    }
